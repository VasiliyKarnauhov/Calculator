const readline = require('readline'); function discription() { console.log('Добро пожаловать!') console.log('Это калькулятор') console.log('Давай начнём работать') } const calculator = { sum(a,b) { return a+b }, minus(a,b) { return a-b }, multiply(a,b) { return a*b }, divide(a,b) { if(b === 0) { console.log('Ошибка деления на ноль') return null } return a/b }, stay(a,b) { return a%b } } discription(); const rl = readline.createInterface({ input: process.stdin, output: process.stdout }); function promptUser() { rl.question('Выбери операцию(sum, minus, multiply, divide, stay): ', (process) => { rl.question('Введи первое число: ', (input1) => { const num1 = parseFloat(input1); rl.question('Введи второе число: ', (input2) => { const num2 = parseFloat(input2); let result; switch (process) { case 'sum': result = calculator.sum(num1,num2) break; case 'minus': result = calculator.minus(num1,num2) break; case 'multiply': result = calculator.multiply(num1,num2) break; case 'divide': result = calculator.divide(num1,num2) break; case 'stay': result = calculator.stay(num1,num2) break; } console.log(`Твой результат: ${result}`); rl.close(); }); }); }); }; promptUser();
